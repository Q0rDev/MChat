145  public static String createBasicBar(float currentValue, float maxValue, float barLength) {
146  int fill = Math.round((currentValue / maxValue) * barLength);
147 
148  String barColor = (fill <= (barLength / 4)) ? "&4" : (fill <= (barLength / 7)) ? "&e" : "&2";
149 
150  StringBuilder out = new StringBuilder();
151  out.append(barColor);
152 
153  for (int i = 0; i < barLength; i++) {
154  if (i == fill)
155  out.append("&8");
156 
157  out.append("|");
158  }
159 
160  out.append("&f");
161 
162  return out.toString();
163  }
